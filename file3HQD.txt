#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main() {
	const string days[] = { "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday" };
	const string subjects[] = { "Math", "Physics", "Chemistry", "Biology", "History" };

	ofstream scheduleFile("schedule.json");
	if (!scheduleFile.is_open()) {
		cerr << "Error: Unable to open output file." << endl;
		return 1;
	}

	scheduleFile << "{\n";
	for (int i = 0; i < 7; ++i) {
		scheduleFile << " \"" << days[i] << "\": [\n";
		for (int j = 0; j < 5; ++j) {
			scheduleFile << " \"" << subjects[j] << "\"";
			if (j < 4) {
				scheduleFile << ",\n";
			}
		}
		scheduleFile << " ]\n";
	}
	scheduleFile << "}\n";

	scheduleFile.close();
	cout << "Schedule written to 'schedule.json'" << endl;

	return 0;
}
